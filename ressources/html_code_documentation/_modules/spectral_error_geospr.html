    P_axial_fdtd, f_ax_fdtd = zip(*(basic_fft(p_axial_fdtd[i, :], t, Ts, 0, 'hanning')

    for i in range(max_freq_idx):

        for j in range(P_axial_fdtd.shape[0]):
